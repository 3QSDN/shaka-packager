33 // These methods work together to notify that the media is encrypted.
34 // If OnEncryptionInfoReady() is called before the media has been started, then
35 // the information is stored and handled when OnEncryptionStart() is called.
36 // if OnEncryptionStart() is called before the media has been started then
37 // OnMediaStart() is responsible for notifying that the segments are encrypted
38 // right away i.e. call OnEncryptionStart().
39 // For now (because Live HLS is not implemented yet) this should be called once,
40 // before media is started. So the logic after the first if statement should not
41 // be taken.
42 void HlsNotifyMuxerListener::OnEncryptionInfoReady(
43  bool is_initial_encryption_info,
44  FourCC protection_scheme,
45  const std::vector<uint8_t>& key_id,
46  const std::vector<uint8_t>& iv,
47  const std::vector<ProtectionSystemSpecificInfo>& key_system_infos) {
48  if (!media_started_) {
49  next_key_id_ = key_id;
50  next_iv_ = iv;
51  next_key_system_infos_ = key_system_infos;
52  return;
53  }
54  for (const ProtectionSystemSpecificInfo& info : key_system_infos) {
55  const bool result = hls_notifier_->NotifyEncryptionUpdate(
56  stream_id_, key_id, info.system_id(), iv, info.pssh_data());
57  LOG_IF(WARNING, !result) << "Failed to add encryption info.";
58  }
59 }

61 void HlsNotifyMuxerListener::OnEncryptionStart() {
62  if (!media_started_) {
63  must_notify_encryption_start_ = true;
64  return;
65  }
66  if (next_key_id_.empty()) {
67  DCHECK(next_iv_.empty());
68  DCHECK(next_key_system_infos_.empty());
69  return;
70  }
71 
72  for (const ProtectionSystemSpecificInfo& info : next_key_system_infos_) {
73  const bool result = hls_notifier_->NotifyEncryptionUpdate(
74  stream_id_, next_key_id_, info.system_id(), next_iv_,
75  info.pssh_data());
76  LOG_IF(WARNING, !result) << "Failed to add encryption info";
77  }
78  next_key_id_.clear();
79  next_iv_.clear();
80  next_key_system_infos_.clear();
81  must_notify_encryption_start_ = false;
82 }

84 void HlsNotifyMuxerListener::OnMediaStart(const MuxerOptions& muxer_options,
85  const StreamInfo& stream_info,
86  uint32_t time_scale,
87  ContainerType container_type) {
88  MediaInfo media_info;
89  if (!internal::GenerateMediaInfo(muxer_options, stream_info, time_scale,
90  container_type, &media_info)) {
91  LOG(ERROR) << "Failed to generate MediaInfo from input.";
92  return;
93  }
94  const bool result = hls_notifier_->NotifyNewStream(
95  media_info, playlist_name_, ext_x_media_name_, ext_x_media_group_id_,
96  &stream_id_);
97  if (!result) {
98  LOG(WARNING) << "Failed to notify new stream.";
99  return;
100  }
101 
102  media_started_ = true;
103  if (must_notify_encryption_start_) {
104  OnEncryptionStart();
105  }
106 }
