28 bool H264ByteToUnitStreamConverter::ConvertByteStreamToNalUnitStream(
29  const uint8_t* input_frame,
30  size_t input_frame_size,
31  std::vector<uint8_t>* output_frame) {
32  DCHECK(input_frame);
33  DCHECK(output_frame);
34 
35  BufferWriter output_buffer(input_frame_size + kStreamConversionOverhead);
36 
37  Nalu nalu;
38  NaluReader reader(NaluReader::kH264, kIsAnnexbByteStream, input_frame,
39  input_frame_size);
40  if (!reader.StartsWithStartCode()) {
41  LOG(ERROR) << "H.264 byte stream frame did not begin with start code.";
42  return false;
43  }
44  while (reader.Advance(&nalu) == NaluReader::kOk) {
45  ProcessNalu(nalu, &output_buffer);
46  }
47 
48  output_buffer.SwapBuffer(output_frame);
49  return true;
50 }

52 void H264ByteToUnitStreamConverter::ProcessNalu(const Nalu& nalu,
53  BufferWriter* output_buffer) {
54  DCHECK(nalu.data());
55  DCHECK(output_buffer);
56 
57  // Skip the start code, but keep the 1-byte NALU type.
58  const uint8_t* nalu_ptr = nalu.data();
59  const uint64_t nalu_size = nalu.payload_size() + nalu.header_size();
60  DCHECK_LE(nalu_size, std::numeric_limits<uint32_t>::max());
61 
62  switch (nalu.type()) {
63  case Nalu::H264_SPS:
64  // Grab SPS NALU.
65  last_sps_.assign(nalu_ptr, nalu_ptr + nalu_size);
66  return;
67  case Nalu::H264_PPS:
68  // Grab PPS NALU.
69  last_pps_.assign(nalu_ptr, nalu_ptr + nalu_size);
70  return;
71  case Nalu::H264_AUD:
72  // Ignore AUD NALU.
73  return;
74  default:
75  // Copy all other NALUs.
76  break;
77  }
78 
79  // Append 4-byte length and NAL unit data to the buffer.
80  output_buffer->AppendInt(static_cast<uint32_t>(nalu_size));
81  output_buffer->AppendArray(nalu_ptr, nalu_size);
82 }

84 bool H264ByteToUnitStreamConverter::GetAVCDecoderConfigurationRecord(
85  std::vector<uint8_t>* decoder_config) {
86  DCHECK(decoder_config);
87 
88  if ((last_sps_.size() < 4) || last_pps_.empty()) {
89  // No data available to construct AVCDecoderConfigurationRecord.
90  return false;
91  }
92 
93  // Construct an AVCDecoderConfigurationRecord containing a single SPS and a
94  // single PPS NALU. Please refer to ISO/IEC 14496-15 for format specifics.
95  BufferWriter buffer(last_sps_.size() + last_pps_.size() + 11);
96  uint8_t version(1);
97  buffer.AppendInt(version);
98  buffer.AppendInt(last_sps_[1]);
99  buffer.AppendInt(last_sps_[2]);
100  buffer.AppendInt(last_sps_[3]);
101  uint8_t reserved_and_length_size_minus_one(0xff);
102  buffer.AppendInt(reserved_and_length_size_minus_one);
103  uint8_t reserved_and_num_sps(0xe1);
104  buffer.AppendInt(reserved_and_num_sps);
105  buffer.AppendInt(static_cast<uint16_t>(last_sps_.size()));
106  buffer.AppendVector(last_sps_);
107  uint8_t num_pps(1);
108  buffer.AppendInt(num_pps);
109  buffer.AppendInt(static_cast<uint16_t>(last_pps_.size()));
110  buffer.AppendVector(last_pps_);
111  buffer.SwapBuffer(decoder_config);
112 
113  return true;
114 }
