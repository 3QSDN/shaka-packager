19 bool H264VideoSliceHeaderParser::Initialize(
20  const std::vector<uint8_t>& decoder_configuration) {
21  // See ISO 14496-15 sec 5.3.3.1.2
22  BufferReader reader(decoder_configuration.data(),
23  decoder_configuration.size());
24  RCHECK(reader.SkipBytes(5));
25 
26  uint8_t sps_count;
27  RCHECK(reader.Read1(&sps_count));
28  sps_count = sps_count & 0x1f;
29 
30  for (size_t i = 0; i < sps_count; i++) {
31  uint16_t size;
32  RCHECK(reader.Read2(&size));
33  const uint8_t* data = reader.data() + reader.pos();
34  RCHECK(reader.SkipBytes(size));
35 
36  int id;
37  Nalu nalu;
38  RCHECK(nalu.InitializeFromH264(data, size));
39  RCHECK(parser_.ParseSPS(nalu, &id) == H264Parser::kOk);
40  }
41 
42  uint8_t pps_count;
43  RCHECK(reader.Read1(&pps_count));
44  for (size_t i = 0; i < pps_count; i++) {
45  uint16_t size;
46  RCHECK(reader.Read2(&size));
47  const uint8_t* data = reader.data() + reader.pos();
48  RCHECK(reader.SkipBytes(size));
49 
50  int id;
51  Nalu nalu;
52  RCHECK(nalu.InitializeFromH264(data, size));
53  RCHECK(parser_.ParsePPS(nalu, &id) == H264Parser::kOk);
54  }
55 
56  return true;
57 }

59 int64_t H264VideoSliceHeaderParser::GetHeaderSize(const Nalu& nalu) {
60  DCHECK(nalu.is_video_slice());
61  H264SliceHeader slice_header;
62  if (parser_.ParseSliceHeader(nalu, &slice_header) != H264Parser::kOk)
63  return -1;
64 
65  // Round-up to bytes.
66  return (slice_header.header_bit_size - 1) / 8 + 1;
67 }
