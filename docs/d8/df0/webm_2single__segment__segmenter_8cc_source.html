29 Status SingleSegmentSegmenter::DoFinalize() {
30  if (!cluster()->Finalize())

32 
33  // Write the Cues to the end of the file.
34  index_start_ = writer_->Position();
35  seek_head()->set_cues_pos(index_start_ - segment_payload_pos());
36  if (!cues()->Write(writer_.get()))
37  return Status(error::FILE_FAILURE, "Error writing Cues data.");
38 
39  uint64_t file_size = writer_->Position();
40  writer_->Position(0);
41 
42  Status status = WriteSegmentHeader(file_size, writer_.get());
43  writer_->Position(file_size);
44  return status;
45 }

51 Status SingleSegmentSegmenter::NewSegment(uint64_t start_timescale) {
52  if (cluster() && !cluster()->Finalize())
53  return Status(error::FILE_FAILURE, "Error finalizing cluster.");
54 
55  // Create a new Cue point.
56  uint64_t position = writer_->Position();
57  uint64_t start_webm_timecode = FromBMFFTimescale(start_timescale);
58 
59  mkvmuxer::CuePoint* cue_point = new mkvmuxer::CuePoint;
60  cue_point->set_time(start_webm_timecode);
61  cue_point->set_track(track_id());
62  cue_point->set_cluster_pos(position - segment_payload_pos());
63  if (!cues()->AddCue(cue_point))
64  return Status(error::INTERNAL_ERROR, "Error adding CuePoint.");
65 
66  return SetCluster(start_webm_timecode, position, writer_.get());
67 }
