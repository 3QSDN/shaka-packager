30 // Inserts emulation byte where necessary.
31 void EscapeRawByteSequencePayload(const uint8_t* input,
32  size_t input_size,
33  BufferWriter* output_writer) {
34  // Keep track of consecutive zeros that it has seen (not including the current
35  // byte), so that the algorithm doesn't need to go back to check the same
36  // bytes.
37  int consecutive_zero_count = 0;
38  for (size_t i = 0; i < input_size; ++i) {
39  if (consecutive_zero_count <= 1) {
40  output_writer->AppendInt(input[i]);
41  } else if (consecutive_zero_count == 2) {
42  if (input[i] == 0 || input[i] == 1 || input[i] == 2 || input[i] == 3) {
43  // Must be escaped.
44  output_writer->AppendInt(kEmulationPreventionByte);
45  }
46  output_writer->AppendInt(input[i]);
47  // Note that input[i] can be 0.
48  // 00 00 00 00 00 00 should become
49  // 00 00 03 00 00 03 00 00 03
50  // So consecutive_zero_count is reset here and incremented below if
51  // input[i] is 0.
52  consecutive_zero_count = 0;
53  }
54 
55  consecutive_zero_count = input[i] == 0 ? consecutive_zero_count + 1 : 0;
56  }
57 
58  // ISO 14496-10 Section 7.4.1.1 mentions that if the last byte is 0 (which
59  // only happens if RBSP has cabac_zero_word), 0x03 must be appended.
60  if (consecutive_zero_count > 0) {
61  DCHECK_GT(input_size, 0u);
62  DCHECK_EQ(input[input_size - 1], 0u);
63  output_writer->AppendInt(kEmulationPreventionByte);
64  }
65 }

92 bool NalUnitToByteStreamConverter::Initialize(
93  const uint8_t* decoder_configuration_data,
94  size_t decoder_configuration_data_size,
95  bool escape_data) {
96  escape_data_ = escape_data;
97  if (!decoder_configuration_data || decoder_configuration_data_size == 0) {
98  LOG(ERROR) << "Decoder conguration is empty.";
99  return false;
100  }
101 
102  AVCDecoderConfiguration decoder_config;
103  if (!decoder_config.Parse(std::vector<uint8_t>(
104  decoder_configuration_data,
105  decoder_configuration_data + decoder_configuration_data_size))) {
106  return false;
107  }
108 
109  if (decoder_config.nalu_count() < 2) {
110  LOG(ERROR) << "Cannot find SPS or PPS.";
111  return false;
112  }
113 
114  nalu_length_size_ = decoder_config.nalu_length_size();
115 
116  BufferWriter buffer_writer(decoder_configuration_data_size);
117  bool found_sps = false;
118  bool found_pps = false;
119  for (uint32_t i = 0; i < decoder_config.nalu_count(); ++i) {
120  const Nalu& nalu = decoder_config.nalu(i);
121  if (nalu.type() == Nalu::H264NaluType::H264_SPS) {
122  buffer_writer.AppendArray(kNaluStartCode, arraysize(kNaluStartCode));
123  AppendNalu(nalu, nalu_length_size_, escape_data, &buffer_writer);
124  found_sps = true;
125  } else if (nalu.type() == Nalu::H264NaluType::H264_PPS) {
126  buffer_writer.AppendArray(kNaluStartCode, arraysize(kNaluStartCode));
127  AppendNalu(nalu, nalu_length_size_, escape_data, &buffer_writer);
128  found_pps = true;
129  }
130  }
131  if (!found_sps || !found_pps) {
132  LOG(ERROR) << "Failed to find SPS or PPS.";
133  return false;
134  }
135 
136  buffer_writer.SwapBuffer(&decoder_configuration_in_byte_stream_);
137  return true;
138 }

140 // This ignores all AUD, SPS, and PPS in the sample. Instead uses the data
141 // parsed in Initialize().
142 bool NalUnitToByteStreamConverter::ConvertUnitToByteStream(
143  const uint8_t* sample,
144  size_t sample_size,
145  bool is_key_frame,
146  std::vector<uint8_t>* output) {
147  if (!sample || sample_size == 0) {
148  LOG(WARNING) << "Sample is empty.";
149  return true;
150  }
151 
152  BufferWriter buffer_writer(sample_size);
153  buffer_writer.AppendArray(kNaluStartCode, arraysize(kNaluStartCode));
154  AddAccessUnitDelimiter(&buffer_writer);
155  if (is_key_frame)
156  buffer_writer.AppendVector(decoder_configuration_in_byte_stream_);
157 
158  NaluReader nalu_reader(NaluReader::kH264, nalu_length_size_, sample,
159  sample_size);
160  Nalu nalu;
161  NaluReader::Result result = nalu_reader.Advance(&nalu);
162 
163  while (result == NaluReader::kOk) {
164  switch (nalu.type()) {
165  case Nalu::H264_AUD:
166  FALLTHROUGH_INTENDED;
167  case Nalu::H264_SPS:
168  FALLTHROUGH_INTENDED;
169  case Nalu::H264_PPS:
170  break;
171  default:
172  buffer_writer.AppendArray(kNaluStartCode, arraysize(kNaluStartCode));
173  AppendNalu(nalu, nalu_length_size_, escape_data_, &buffer_writer);
174  break;
175  }
176  result = nalu_reader.Advance(&nalu);
177  }
178 
179  DCHECK_NE(result, NaluReader::kOk);
180  if (result != NaluReader::kEOStream) {
181  LOG(ERROR) << "Stopped reading before end of stream.";
182  return false;
183  }
184 
185  buffer_writer.SwapBuffer(output);
186  return true;
187 }
