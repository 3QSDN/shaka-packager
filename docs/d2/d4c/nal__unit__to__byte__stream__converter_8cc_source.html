50 bool CheckIsClearNalu(const std::vector<SubsampleEntry>* subsamples,
51  size_t subsample_id,
52  size_t nalu_size,
53  bool* is_nalu_all_clear) {
54  if (subsample_id >= subsamples->size()) {
55  LOG(ERROR) << "Subsample index exceeds subsamples' size.";
56  return false;
57  }
58  const SubsampleEntry& subsample = subsamples->at(subsample_id);
59  if (nalu_size == subsample.clear_bytes + subsample.cipher_bytes) {
60  *is_nalu_all_clear = false;
61  } else if (nalu_size < subsample.clear_bytes) {
62  *is_nalu_all_clear = true;
63  } else {
64  LOG(ERROR) << "Unexpected subsample entry " << subsample.clear_bytes << ":"
65  << subsample.cipher_bytes << " nalu size: " << nalu_size;
66  return false;
67  }
68  return true;
69 }

73 void EscapeNalByteSequence(const uint8_t* input,
74  size_t input_size,
75  BufferWriter* output_writer) {
76  // Keep track of consecutive zeros that it has seen (not including the current
77  // byte), so that the algorithm doesn't need to go back to check the same
78  // bytes.
79  int consecutive_zero_count = 0;
80  for (size_t i = 0; i < input_size; ++i) {
81  if (consecutive_zero_count <= 1) {
82  output_writer->AppendInt(input[i]);
83  } else if (consecutive_zero_count == 2) {
84  if (input[i] == 0 || input[i] == 1 || input[i] == 2 || input[i] == 3) {
85  // Must be escaped.
86  output_writer->AppendInt(kEmulationPreventionByte);
87  }
88 
89  output_writer->AppendInt(input[i]);
90  // Note that input[i] can be 0.
91  // 00 00 00 00 00 00 should become
92  // 00 00 03 00 00 03 00 00 03
93  // So consecutive_zero_count is reset here and incremented below if
94  // input[i] is 0.
95  consecutive_zero_count = 0;
96  }
97 
98  consecutive_zero_count = input[i] == 0 ? consecutive_zero_count + 1 : 0;
99  }
100 
101  // ISO 14496-10 Section 7.4.1.1 mentions that if the last byte is 0 (which
102  // only happens if RBSP has cabac_zero_word), 0x03 must be appended.
103  if (consecutive_zero_count > 0) {
104  DCHECK_GT(input_size, 0u);
105  DCHECK_EQ(input[input_size - 1], 0u);
106  output_writer->AppendInt(kEmulationPreventionByte);
107  }
108 }

114 bool NalUnitToByteStreamConverter::Initialize(
115  const uint8_t* decoder_configuration_data,
116  size_t decoder_configuration_data_size) {
117  if (!decoder_configuration_data || decoder_configuration_data_size == 0) {
118  LOG(ERROR) << "Decoder conguration is empty.";
119  return false;
120  }
121 
122  AVCDecoderConfigurationRecord decoder_config;
123  if (!decoder_config.Parse(std::vector<uint8_t>(
124  decoder_configuration_data,
125  decoder_configuration_data + decoder_configuration_data_size))) {
126  return false;
127  }
128 
129  if (decoder_config.nalu_count() < 2) {
130  LOG(ERROR) << "Cannot find SPS or PPS.";
131  return false;
132  }
133 
134  nalu_length_size_ = decoder_config.nalu_length_size();
135 
136  BufferWriter buffer_writer(decoder_configuration_data_size);
137  bool found_sps = false;
138  bool found_pps = false;
139  for (uint32_t i = 0; i < decoder_config.nalu_count(); ++i) {
140  const Nalu& nalu = decoder_config.nalu(i);
141  if (nalu.type() == Nalu::H264NaluType::H264_SPS) {
142  buffer_writer.AppendArray(kNaluStartCode, arraysize(kNaluStartCode));
143  AppendNalu(nalu, nalu_length_size_, !kEscapeData, &buffer_writer);
144  found_sps = true;
145  } else if (nalu.type() == Nalu::H264NaluType::H264_PPS) {
146  buffer_writer.AppendArray(kNaluStartCode, arraysize(kNaluStartCode));
147  AppendNalu(nalu, nalu_length_size_, !kEscapeData, &buffer_writer);
148  found_pps = true;
149  }
150  }
151  if (!found_sps || !found_pps) {
152  LOG(ERROR) << "Failed to find SPS or PPS.";
153  return false;
154  }
155 
156  buffer_writer.SwapBuffer(&decoder_configuration_in_byte_stream_);
157  return true;
158 }

172 // This ignores all AUD, SPS, and PPS in the sample. Instead uses the data
173 // parsed in Initialize().
174 bool NalUnitToByteStreamConverter::ConvertUnitToByteStreamWithSubsamples(
175  const uint8_t* sample,
176  size_t sample_size,
177  bool is_key_frame,
178  bool escape_encrypted_nalu,
179  std::vector<uint8_t>* output,
180  std::vector<SubsampleEntry>* subsamples) {
181  if (!sample || sample_size == 0) {
182  LOG(WARNING) << "Sample is empty.";
183  return true;
184  }
185 
186  BufferWriter buffer_writer(sample_size);
187  buffer_writer.AppendArray(kNaluStartCode, arraysize(kNaluStartCode));
188  AddAccessUnitDelimiter(&buffer_writer);
189  if (is_key_frame)
190  buffer_writer.AppendVector(decoder_configuration_in_byte_stream_);
191 
192  int adjustment = static_cast<int>(buffer_writer.Size());
193  const int start_code_size_adjustment =
194  arraysize(kNaluStartCode) - nalu_length_size_;
195  size_t subsample_id = 0;
196 
197  NaluReader nalu_reader(Nalu::kH264, nalu_length_size_, sample, sample_size);
198  Nalu nalu;
199  NaluReader::Result result = nalu_reader.Advance(&nalu);
200 
201  while (result == NaluReader::kOk) {
202  switch (nalu.type()) {
203  case Nalu::H264_AUD:
204  FALLTHROUGH_INTENDED;
205  case Nalu::H264_SPS:
206  FALLTHROUGH_INTENDED;
207  case Nalu::H264_PPS:
208  if (subsamples && !subsamples->empty()) {
209  const size_t old_nalu_size =
210  nalu_length_size_ + nalu.header_size() + nalu.payload_size();
211  bool is_nalu_all_clear;
212  if (!CheckIsClearNalu(subsamples, subsample_id, old_nalu_size,
213  &is_nalu_all_clear)) {
214  return false;
215  }
216  if (is_nalu_all_clear) {
217  // If AUD/SPS/PPS is all clear, reduce the clear bytes.
218  DCHECK_LT(old_nalu_size, subsamples->at(subsample_id).clear_bytes);
219  subsamples->at(subsample_id).clear_bytes -=
220  static_cast<uint16_t>(old_nalu_size);
221  } else {
222  // If AUD/SPS/PPS has cipher, drop the corresponding subsample.
223  subsamples->erase(subsamples->begin() + subsample_id);
224  }
225  }
226  break;
227  default:
228  bool escape_data = false;
229  if (subsamples && !subsamples->empty()) {
230  const size_t old_nalu_size =
231  nalu_length_size_ + nalu.header_size() + nalu.payload_size();
232  bool is_nalu_all_clear;
233  if (!CheckIsClearNalu(subsamples, subsample_id, old_nalu_size,
234  &is_nalu_all_clear)) {
235  return false;
236  }
237  if (is_nalu_all_clear) {
238  // Add this nalu to the adjustment and remove it from clear_bytes.
239  DCHECK_LT(old_nalu_size, subsamples->at(subsample_id).clear_bytes);
240  subsamples->at(subsample_id).clear_bytes -=
241  static_cast<uint16_t>(old_nalu_size);
242  adjustment +=
243  static_cast<int>(old_nalu_size) + start_code_size_adjustment;
244  } else {
245  if (escape_encrypted_nalu)
246  escape_data = subsamples->at(subsample_id).cipher_bytes != 0;
247  // Apply the adjustment on the current subsample, reset the
248  // adjustment and move to the next subsample.
249  subsamples->at(subsample_id).clear_bytes +=
250  adjustment + start_code_size_adjustment;
251  subsample_id++;
252  adjustment = 0;
253  }
254  }
255  buffer_writer.AppendArray(kNaluStartCode, arraysize(kNaluStartCode));
256  AppendNalu(nalu, nalu_length_size_, escape_data, &buffer_writer);
257  break;
258  }
259 
260  result = nalu_reader.Advance(&nalu);
261  }
262 
263  DCHECK_NE(result, NaluReader::kOk);
264  if (result != NaluReader::kEOStream) {
265  LOG(ERROR) << "Stopped reading before end of stream.";
266  return false;
267  }
268 
269  buffer_writer.SwapBuffer(output);
270  return true;
271 }
