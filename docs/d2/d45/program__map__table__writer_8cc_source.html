112 // Note there are dozens of CRCs. This is one of them.
113 // http://reveng.sourceforge.net/crc-catalogue/all.htm
114 uint32_t Crc32Mpeg2(const uint8_t* data, size_t data_size) {
115  uint32_t crc = 0xFFFFFFFF;
116  for (size_t i = 0; i < data_size; ++i) {
117  crc = kCrcTable[((crc >> 24) ^ data[i]) & 0xFF] ^ (crc << 8);
118  }
119  return crc;
120 }

200 bool WriteAacAudioSetupInformation(const uint8_t* aac_audio_specific_config,
201  size_t aac_audio_specific_config_size,
202  BufferWriter* audio_setup_information) {
203  AACAudioSpecificConfig config;
204  const bool result = config.Parse(std::vector<uint8_t>(
205  aac_audio_specific_config,
206  aac_audio_specific_config + aac_audio_specific_config_size));
207  if (!result) {
208  LOG(WARNING) << "Failed to parse config. Assuming AAC-LC.";
209  return false;
210  }
211 
212  const uint8_t kAacLc = 2;
213  const uint8_t kAacHeV1 = 5;
214  const uint8_t kAacHeV2 = 29;
215  uint8_t audio_object_type = 2;
216  audio_object_type = config.audio_object_type();
217 
218  switch (audio_object_type) {
219  case kAacLc:
220  audio_setup_information->AppendInt(FOURCC_zaac);
221  break;
222  case kAacHeV1:
223  audio_setup_information->AppendInt(FOURCC_zach);
224  break;
225  case kAacHeV2:
226  audio_setup_information->AppendInt(FOURCC_zacp);
227  break;
228  default:
229  LOG(ERROR) << "Unknown object type for aac " << audio_object_type;
230  return false;
231  }
232 
233  // Priming. Since no info from encoder, set it to 0x0000.
234  audio_setup_information->AppendInt(static_cast<uint16_t>(0x0000));
235  // Version is always 0x01.
236  audio_setup_information->AppendInt(static_cast<uint8_t>(0x01));
237  audio_setup_information->AppendInt(
238  static_cast<uint8_t>(aac_audio_specific_config_size));
239  audio_setup_information->AppendArray(aac_audio_specific_config,
240  aac_audio_specific_config_size);
241  return true;
242 }

264 void WritePmtWithParameters(uint8_t stream_type,
265  int version,
266  int current_next_indicator,
267  const uint8_t* descriptors,
268  size_t descriptors_size,
269  ContinuityCounter* continuity_counter,
270  BufferWriter* output) {
271  DCHECK(current_next_indicator == kCurrent || current_next_indicator == kNext);
272  // Body starting from program number.
273  BufferWriter pmt_body;
274  pmt_body.AppendInt(static_cast<uint8_t>(0x00));
275  pmt_body.AppendInt(kProgramNumber);
276  // resevered bits then version and current_next_indicator.
277  pmt_body.AppendInt(
278  static_cast<uint8_t>(0xC0 |
279  static_cast<uint8_t>(version) << 1 |
280  static_cast<uint8_t>(current_next_indicator)));
281  // section number.
282  pmt_body.AppendInt(static_cast<uint8_t>(0x00));
283  // last section number.
284  pmt_body.AppendInt(static_cast<uint8_t>(0x00));
285  // first 3 bits reserved. Rest is unused bits for PCR PID.
286  pmt_body.AppendInt(static_cast<uint8_t>(0xE0));
287  pmt_body.AppendInt(ProgramMapTableWriter::kElementaryPid);
288  // First 4 bits are reserved. Next 12 bits is program_info_length which is 0.
289  pmt_body.AppendInt(static_cast<uint8_t>(0xF0));
290  pmt_body.AppendInt(static_cast<uint8_t>(0x00));
291 
292  pmt_body.AppendInt(stream_type);
293  // 3 reserved bits followed by 13 bit elementary_PID.
294  pmt_body.AppendInt(static_cast<uint8_t>(0xE0));
295  pmt_body.AppendInt(ProgramMapTableWriter::kElementaryPid);
296 
297  // 4 reserved bits followed by ES_info_length.
298  pmt_body.AppendInt(static_cast<uint16_t>(0xF000 | descriptors_size));
299  pmt_body.AppendArray(descriptors, descriptors_size);
300 
301  // The whole PMT has 3 bytes before the body and 4 more bytes for CRC. This
302  // also includes pointer field (1 byte) so + 8 in total.
303  BufferWriter pmt(pmt_body.Size() + 8);
304  // Pointer field.
305  pmt.AppendInt(static_cast<uint8_t>(0x00));
306  // PMT table ID is always 2.
307  pmt.AppendInt(static_cast<uint8_t>(0x02));
308  // First four bits must be '1011'. +4 for CRC.
309  pmt.AppendInt(static_cast<uint16_t>(0xB000 | (pmt_body.Size() + 4)));
310  pmt.AppendBuffer(pmt_body);
311 
312  // Don't include the pointer field.
313  const uint32_t crc = Crc32Mpeg2(pmt.Buffer() + 1, pmt.Size() - 1);
314  pmt.AppendInt(crc);
315  WritePmtToBuffer(pmt.Buffer(), pmt.Size(), continuity_counter, output);
316 }
